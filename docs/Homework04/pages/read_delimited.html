Continue chain: grouped means & medians for numeric vars by class

yeast_filer_data <- yeast |>
  select(-c(seq_name, nuc)) |>                    
  group_by(class) |>                              
  mutate(
    across(
      .cols = c(mcg, gvh, alm, mit, erl, pox, vac), 
      .fns  = list(mean = ~ mean(.x, na.rm = TRUE),
                   median = ~ median(.x, na.rm = TRUE)),
      .names = "{.col}_{.fn}"
    )
  ) |>
  ungroup()
